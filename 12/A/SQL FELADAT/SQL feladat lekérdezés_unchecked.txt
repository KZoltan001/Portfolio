Sorold fel a tesztelőket!/
SELECT f.Felh_id, f.Felhnev, f.Email
FROM Felhasznalok f
JOIN Felhasznaloi_Szerepkor fs ON f.Felh_id = fs.Felh_id
JOIN Szerepkorok s ON fs.Egyedi_id = s.Egyedi_id
WHERE s.Szerepkor_nev = 'Tesztelő';


Cseréld ki a Szállító szerepkört Szállítók névre maria felhasználónévhez ki tartozik?/
-- Az 'maria' felhasználóhoz tartozó szerepkör módosítása
UPDATE Szerepkorok
SET Szerepkor_nev = 'Szállítók'
WHERE Szerepkor_nev = 'Szállító' 
AND EXISTS (
    SELECT 1
    FROM Felhasznaloi_Szerepkor fs
    JOIN Felhasznalok f ON fs.Felh_id = f.Felh_id
    WHERE f.Felhnev = 'maria'
);
